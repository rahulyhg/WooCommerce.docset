 31     /**
 32      * Constructor for the session class.
 33      *
 34      * @access public
 35      * @return void
 36      */
 37     public function __construct() {
 38         $this->_cookie = 'wp_woocommerce_session_' . COOKIEHASH;
 39 
 40         if ( $cookie = $this->get_session_cookie() ) {
 41             $this->_customer_id        = $cookie[0];
 42             $this->_session_expiration = $cookie[1];
 43             $this->_session_expiring   = $cookie[2];
 44 
 45             // Update session if its close to expiring
 46             if ( time() > $this->_session_expiring ) {
 47                 $this->set_session_expiration();
 48                 $session_expiry_option = '_wc_session_expires_' . $this->_customer_id;
 49                 // Check if option exists first to avoid auloading cleaned up sessions
 50                 if ( false === get_option( $session_expiry_option ) ) {
 51                     add_option( $session_expiry_option, $this->_session_expiration, '', 'no' );
 52                 } else {
 53                     update_option( $session_expiry_option, $this->_session_expiration );
 54                 }
 55             }
 56 
 57         } else {
 58             $this->set_session_expiration();
 59             $this->_customer_id = $this->generate_customer_id();
 60         }
 61 
 62         $this->_data = $this->get_session_data();
 63 
 64         // Actions
 65         add_action( 'woocommerce_set_cart_cookies', array( $this, 'set_customer_session_cookie' ), 10 );
 66         add_action( 'woocommerce_cleanup_sessions', array( $this, 'cleanup_sessions' ), 10 );
 67         add_action( 'shutdown', array( $this, 'save_data' ), 20 );
 68     }

 70     /**
 71      * Sets the session cookie on-demand (usually after adding an item to the cart).
 72      *
 73      * Since the cookie name (as of 2.1) is prepended with wp, cache systems like batcache will not cache pages when set.
 74      *
 75      * Warning: Cookies will only be set if this is called before the headers are sent.
 76      */
 77     public function set_customer_session_cookie( $set ) {
 78         if ( $set ) {
 79             // Set/renew our cookie
 80             $to_hash      = $this->_customer_id . $this->_session_expiration;
 81             $cookie_hash  = hash_hmac( 'md5', $to_hash, wp_hash( $to_hash ) );
 82             $cookie_value = $this->_customer_id . '||' . $this->_session_expiration . '||' . $this->_session_expiring . '||' . $cookie_hash;
 83 
 84             // Set the cookie
 85             wc_setcookie( $this->_cookie, $cookie_value, $this->_session_expiration, apply_filters( 'wc_session_use_secure_cookie', false ) );
 86         }
 87     }

113     /**
114      * get_session_cookie function.
115      *
116      * @access public
117      * @return mixed
118      */
119     public function get_session_cookie() {
120         if ( empty( $_COOKIE[ $this->_cookie ] ) )
121             return false;
122 
123         list( $customer_id, $session_expiration, $session_expiring, $cookie_hash ) = explode( '||', $_COOKIE[ $this->_cookie ] );
124 
125         // Validate hash
126         $to_hash = $customer_id . $session_expiration;
127         $hash    = hash_hmac( 'md5', $to_hash, wp_hash( $to_hash ) );
128 
129         if ( $hash != $cookie_hash )
130             return false;
131 
132         return array( $customer_id, $session_expiration, $session_expiring, $cookie_hash );
133     }

146      * save_data function.
147      *
148      * @access public
149      * @return void
150      */
151     public function save_data() {
152         // Dirty if something changed - prevents saving nothing new
153         if ( $this->_dirty ) {
154 
155             $session_option = '_wc_session_' . $this->_customer_id;
156             $session_expiry_option = '_wc_session_expires_' . $this->_customer_id;
157 
158             if ( false === get_option( $session_option ) ) {
159                 add_option( $session_option, $this->_data, '', 'no' );
160                 add_option( $session_expiry_option, $this->_session_expiration, '', 'no' );
161             } else {
162                 update_option( $session_option, $this->_data );
163             }
164         }

167     /**
168      * cleanup_sessions function.
169      *
170      * @access public
171      * @return void
172      */
173     public function cleanup_sessions() {
174         global $wpdb;
175 
176         if ( ! defined( 'WP_SETUP_CONFIG' ) && ! defined( 'WP_INSTALLING' ) ) {
177             $now                = time();
178             $expired_sessions   = array();
179             $wc_session_expires = $wpdb->get_results( "SELECT option_name, option_value FROM $wpdb->options WHERE option_name LIKE '_wc_session_expires_%'" );
180 
181             foreach ( $wc_session_expires as $wc_session_expire ) {
182                 if ( $now > intval( $wc_session_expire->option_value ) ) {
183                     $session_id         = substr( $wc_session_expire->option_name, 20 );
184                     $expired_sessions[] = $wc_session_expire->option_name;  // Expires key
185                     $expired_sessions[] = "_wc_session_$session_id"; // Session key
186                 }
187             }
188 
189             if ( ! empty( $expired_sessions ) ) {
190                 $expired_sessions_chunked = array_chunk( $expired_sessions, 100 );
191                 foreach ( $expired_sessions_chunked as $chunk ) {
192                     $option_names = implode( "','", $chunk );
193                     $wpdb->query( "DELETE FROM $wpdb->options WHERE option_name IN ('$option_names')" );
194                 }
195             }
196         }
