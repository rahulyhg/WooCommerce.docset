110     /**
111      * Check If The Gateway Is Available For Use
112      *
113      * @return bool
114      */
115     public function is_available() {
116         $order = null;
117 
118         if ( ! $this->enable_for_virtual ) {
119             if ( WC()->cart && ! WC()->cart->needs_shipping() ) {
120                 return false;
121             }
122 
123             if ( is_page( wc_get_page_id( 'checkout' ) ) && 0 < get_query_var( 'order-pay' ) ) {
124                 $order_id = absint( get_query_var( 'order-pay' ) );
125                 $order    = new WC_Order( $order_id );
126 
127                 // Test if order needs shipping.
128                 $needs_shipping = false;
129 
130                 if ( 0 < sizeof( $order->get_items() ) ) {
131                     foreach ( $order->get_items() as $item ) {
132                         $_product = $order->get_product_from_item( $item );
133 
134                         if ( $_product->needs_shipping() ) {
135                             $needs_shipping = true;
136                             break;
137                         }
138                     }
139                 }
140 
141                 $needs_shipping = apply_filters( 'woocommerce_cart_needs_shipping', $needs_shipping );
142 
143                 if ( $needs_shipping ) {
144                     return false;
145                 }
146             }
147         }
148 
149         if ( ! empty( $this->enable_for_methods ) ) {
150 
151             // Only apply if all packages are being shipped via local pickup
152             $chosen_shipping_methods_session = WC()->session->get( 'chosen_shipping_methods' );
153 
154             if ( isset( $chosen_shipping_methods_session ) ) {
155                 $chosen_shipping_methods = array_unique( $chosen_shipping_methods_session );
156             } else {
157                 $chosen_shipping_methods = array();
158             }
159 
160             $check_method = false;
161 
162             if ( is_object( $order ) ) {
163                 if ( $order->shipping_method ) {
164                     $check_method = $order->shipping_method;
165                 }
166 
167             } elseif ( empty( $chosen_shipping_methods ) || sizeof( $chosen_shipping_methods ) > 1 ) {
168                 $check_method = false;
169             } elseif ( sizeof( $chosen_shipping_methods ) == 1 ) {
170                 $check_method = $chosen_shipping_methods[0];
171             }
172 
173             if ( ! $check_method ) {
174                 return false;
175             }
176 
177             $found = false;
178 
179             foreach ( $this->enable_for_methods as $method_id ) {
180                 if ( strpos( $check_method, $method_id ) === 0 ) {
181                     $found = true;
182                     break;
183                 }
184             }
185 
186             if ( ! $found ) {
187                 return false;
188             }
189         }
190 
191         return parent::is_available();
192     }

195     /**
196      * Process the payment and return the result
197      *
198      * @param int $order_id
199      * @return array
200      */
201     public function process_payment( $order_id ) {
202 
203         $order = new WC_Order( $order_id );
204 
205         // Mark as processing (payment won't be taken until delivery)
206         $order->update_status( 'processing', __( 'Payment to be made upon delivery.', 'woocommerce' ) );
207 
208         // Reduce stock levels
209         $order->reduce_order_stock();
210 
211         // Remove cart
212         WC()->cart->empty_cart();
213 
214         // Return thankyou redirect
215         return array(
216             'result'    => 'success',
217             'redirect'  => $this->get_return_url( $order )
218         );
219     }
