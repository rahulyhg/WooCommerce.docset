 89     /**

 93      * @param WC_Order $order
 94      * @param bool $sent_to_admin
 95      * @param bool $plain_text
 96      */
 97     public function email_instructions( $order, $sent_to_admin, $plain_text = false ) {
 98         if ( $sent_to_admin || $order->status !== 'on-hold' || $order->payment_method !== 'cheque' )
 99             return;
100 
101         if ( $this->instructions )
102             echo wpautop( wptexturize( $this->instructions ) );

105     /**

111     public function process_payment( $order_id ) {
112 
113         $order = new WC_Order( $order_id );
114 
115         // Mark as on-hold (we're awaiting the cheque)
116         $order->update_status( 'on-hold', __( 'Awaiting cheque payment', 'woocommerce' ) );
117 
118         // Reduce stock levels
119         $order->reduce_order_stock();
120 
121         // Remove cart
122         WC()->cart->empty_cart();
123 
124         // Return thankyou redirect
125         return array(
126             'result'    => 'success',
127             'redirect'  => $this->get_return_url( $order )
