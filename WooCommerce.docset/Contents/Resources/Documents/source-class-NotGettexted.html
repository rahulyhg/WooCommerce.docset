 55     function list_php_files($dir) {
 56         $files = array();
 57         $items = scandir( $dir );
 58         foreach ( (array) $items as $item ) {
 59             $full_item = $dir . '/' . $item;
 60             if ('.' == $item || '..' == $item)
 61                 continue;
 62             if ('.php' == substr($item, -4))
 63                 $files[] = $full_item;
 64             if (is_dir($full_item))
 65                 $files += array_merge($files, NotGettexted::list_php_files($full_item, $files));
 66         }
 67         return $files;
 68     }

 81     function walk_tokens(&$tokens, $string_action, $other_action, $register_action=null) {
 82 
 83         $current_comment_id = '';
 84         $current_string = '';
 85         $current_string_line = 0;
 86 
 87         $result = '';
 88         $line = 1;
 89 
 90         foreach($tokens as $token) {
 91             if (is_array($token)) {
 92                 list($id, $text) = $token;
 93                 $line += substr_count($text, "\n");
 94                 if ((T_ML_COMMENT == $id || T_COMMENT == $id) && preg_match('|/\*\s*(/?WP_I18N_[a-z_]+)\s*\*/|i', $text, $matches)) {
 95                     if ($this->STAGE_OUTSIDE == $stage) {
 96                         $stage = $this->STAGE_START_COMMENT;
 97                         $current_comment_id = $matches[1];
 98                         $this->logmsg('start comment', $current_comment_id);
 99                         $result .= call_user_func($other_action, $token);
100                         continue;
101                     }
102                     if ($this->STAGE_START_COMMENT <= $stage && $stage <= $this->STAGE_WHITESPACE_AFTER && '/'.$current_comment_id == $matches[1]) {
103                         $stage = $this->STAGE_END_COMMENT;
104                         $this->logmsg('end comment', $current_comment_id);
105                         $result .= call_user_func($other_action, $token);
106                         if (!is_null($register_action)) call_user_func($register_action, $current_string, $current_comment_id, $current_string_line);
107                         continue;
108                     }
109                 } else if (T_CONSTANT_ENCAPSED_STRING == $id) {
110                     if ($this->STAGE_START_COMMENT <= $stage && $stage < $this->STAGE_WHITESPACE_AFTER) {
111                         eval('$current_string='.$text.';');
112                         $this->logmsg('string', $current_string);
113                         $current_string_line = $line;
114                         $result .= call_user_func($string_action, $token, $current_string);
115                         continue;
116                     }
117                 } else if (T_WHITESPACE == $id) {
118                     if ($this->STAGE_START_COMMENT <= $stage && $stage < $this->STAGE_STRING) {
119                         $stage = $this->STAGE_WHITESPACE_BEFORE;
120                         $this->logmsg('whitespace before');
121                         $result .= call_user_func($other_action, $token);
122                         continue;
123                     }
124                     if ($this->STAGE_STRING < $stage && $stage < $this->STAGE_END_COMMENT) {
125                         $stage = $this->STAGE_WHITESPACE_AFTER;
126                         $this->logmsg('whitespace after');
127                         $result .= call_user_func($other_action, $token);
128                         continue;
129                     }
130                 }
131             }
132             $result .= call_user_func($other_action, $token);
133             $stage = $this->STAGE_OUTSIDE;
134             $current_comment_id = '';
135             $current_string = '';
136             $current_string_line = 0;
137         }
138         return $result;
139     }

142     function command_extract() {
143         $args = func_get_args();
144         $pot_filename = $args[0];
145         if (isset($args[1]) && is_array($args[1]))
146             $filenames = $args[1];
147         else
148             $filenames = array_slice($args, 1);
149 
150         $global_name = '__entries_'.mt_rand(1, 1000);
151         $GLOBALS[$global_name] = array();
152 
153         foreach($filenames as $filename) {
154             $tokens = token_get_all(file_get_contents($filename));
155             $aggregator = $this->make_string_aggregator($global_name, $filename);
156             $this->walk_tokens($tokens, array(&$this, 'ignore_token'), array(&$this, 'ignore_token'), $aggregator);
157         }
158 
159         $potf = '-' == $pot_filename? STDOUT : @fopen($pot_filename, 'a');
160         if (false === $potf) {
161             $this->cli_die("Couldn't open pot file: $pot_filename");
162         }
163 
164         foreach($GLOBALS[$global_name] as $item) {
165             @list($string, $comment_id, $filename, $line_number) = $item;
166             $filename = isset($filename)? preg_replace('|^\./|', '', $filename) : '';
167             $ref_line_number = isset($line_number)? ":$line_number" : '';
168             $args = array(
169                 'singular' => $string,
170                 'extracted_comments' => "Not gettexted string $comment_id",
171                 'references' => array("$filename$ref_line_number"),
172             );
173             $entry = new Translation_Entry($args);
174             fwrite($potf, "\n".PO::export_entry($entry)."\n");
175         }
176         if ('-' != $pot_filename) fclose($potf);
177         return true;
178     }

180     function command_replace() {
181         $args = func_get_args();
182         $mo_filename = $args[0];
183         if (isset($args[1]) && is_array($args[1]))
184             $filenames = $args[1];
185         else
186             $filenames = array_slice($args, 1);
187 
188         $global_name = '__mo_'.mt_rand(1, 1000);
189         $GLOBALS[$global_name] = new MO();
190         $replacer = $this->make_mo_replacer($global_name);
191 
192         $res = $GLOBALS[$global_name]->import_from_file($mo_filename);
193         if (false === $res) {
194             $this->cli_die("Couldn't read MO file '$mo_filename'!");
195         }
196         foreach($filenames as $filename) {
197             $source = file_get_contents($filename);
198             if ( strlen($source) > 150000 ) continue;
199             $tokens = token_get_all($source);
200             $new_file = $this->walk_tokens($tokens, $replacer, array(&$this, 'unchanged_token'));
201             $f = fopen($filename, 'w');
202             fwrite($f, $new_file);
203             fclose($f);
204         }
205         return true;
206     }
