33     function load_from_file($ext_filename) {
34         $source = WC_Makepot::get_first_lines($ext_filename);
35         $pot = '';
36         foreach($this->headers as $header) {
37             $string = WC_Makepot::get_addon_header($header, $source);
38             if (!$string) continue;
39             $args = array(
40                 'singular' => $string,
41                 'extracted_comments' => $header.' of the plugin/theme',
42             );
43             $entry = new Translation_Entry($args);
44             $pot .= "\n".PO::export_entry($entry)."\n";
45         }

49     function append( $ext_filename, $pot_filename, $headers = null ) {
50         if ( $headers )
51             $this->headers = (array) $headers;
52         if ( is_dir( $ext_filename ) ) {
53             $pot = implode('', array_map(array(&$this, 'load_from_file'), glob("$ext_filename/*.php")));
54         } else {
55             $pot = $this->load_from_file($ext_filename);
56         }
57         $potf = '-' == $pot_filename? STDOUT : fopen( $pot_filename, 'a' );
58         if (!$potf) return false;
59         fwrite($potf, $pot);
60         if ('-' != $pot_filename) fclose($potf);
61         return true;
62     }
